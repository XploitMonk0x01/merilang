class BankAccount {
    vidhi __init__(balance) {
        yeh.balance = balance
    }
    samapt
    
    vidhi deposit(amount) {
        yeh.balance = yeh.balance + amount
        dikhao "Deposited: " + str(amount)
    }
    samapt
    
    vidhi withdraw(amount) {
        agar amount > yeh.balance {
            dikhao "Insufficient funds!"
            vapas galat
        }
        bas
        yeh.balance = yeh.balance - amount
        vapas sahi
    }
    samapt
}

account = naya BankAccount(1000)
account.deposit(500)
account.withdraw(200)
